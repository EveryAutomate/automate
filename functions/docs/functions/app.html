 32    def read(self, collection: str, identifier: Optional[str] = None, filters: Optional[List] = None) -> List[Dict]:
 33        """
 34        Read documents from a Firestore collection.
 35
 36        :param collection: The Firestore collection name.
 37        :param identifier: Optional document identifier.
 38        :param filters: Optional list of filters as tuples (field, operator, value).
 39        :return: List of documents as dictionaries.
 40        """
 41        collection_ref = self.db.collection(collection)
 42        
 43        if identifier:
 44            doc = collection_ref.document(identifier).get()
 45            return [doc.to_dict()] if doc.exists else []
 46        
 47        if filters:
 48            query = collection_ref
 49            for field, operator, value in filters:
 50                query = query.where(filter=FieldFilter(field, operator, value))
 51            docs = query.stream()
 52        else:
 53            docs = collection_ref.stream()
 54            
 55        return [doc.to_dict() for doc in docs]

 57    def write(self, collection: str, data: Union[Dict, List[Dict]]) -> None:
 58        """
 59        Write new documents to a Firestore collection.
 60
 61        :param collection: The Firestore collection name.
 62        :param data: A single document or list of documents to write.
 63        :raises ValueError: If data does not meet the required structure or has duplicate tags.
 64        """
 65        collection_ref = self.db.collection(collection)
 66        batch = self.db.batch()
 67
 68        documents = [data] if isinstance(data, dict) else data
 69        document_tags_seen = set()
 70
 71        for document in documents:
 72            document_tag = document.get("document_tag")
 73            contents = document.get("contents")
 74
 75            if not document_tag or not contents:
 76                raise ValueError("Each document must have 'document_tag' and 'contents'.")
 77
 78            if document_tag in document_tags_seen:
 79                raise ValueError(f"Duplicate document tag '{document_tag}'")
 80
 81            document_tags_seen.add(document_tag)
 82            doc_ref = collection_ref.document(document_tag)
 83
 84            if doc_ref.get().exists:
 85                raise ValueError(f"Document '{document_tag}' already exists")
 86
 87            batch.set(doc_ref, contents)
 88
 89        batch.commit()
 90        return True

182            payload = {k: data[v] for k, v in config['payload_mapping'].items() if v in data}
183
184            # Gemini-specific adjustments
185            if 'contents' in payload:
186                payload['contents'] = [{"parts": [{"text": payload['contents']}]}]
187
188        # Log the payload
189        import logging
190        logging.basicConfig(level=logging.INFO)
191        logger = logging.getLogger(__name__)
192        logger.info("Final payload: %s", json.dumps(payload, indent=2))
193
194        # Make the API request
195        response = requests.request(
196            method=config['method'],
197            url=config['url'],
198            headers=config.get('headers', {}),
199            params=params,
200            json=payload
201        )
202
203        # Handle errors
204        if response.status_code >= 400:
205            logger.error("Error response body: %s", response.text)
206            raise requests.exceptions.RequestException(
207                f"External service error: {response.status_code} - {response.reason}"
208            )

210        # Return the JSON response
211        return response.json()

213    def manipulate(self, method_name: str, inputs: Dict) -> Any:
214        """
215        Execute a data manipulation process based on stored configuration.
216
217        :param method_name: Name of the manipulation method.
218        :param inputs: Input parameters for the process.
219        :return: Result of the manipulation process.
220        :raises ValueError: If configuration or inputs are invalid.
221        """
222        # Get process model from firestore
223        process_docs = self.read(collection=method_name)
224        
225        if not process_docs:
226            raise ValueError(f"Method {method_name} not found")
227        
228        # Get the process document
229        process_doc = process_docs[0]
230        
231        # Validate process document structure
232        if 'output_key' not in process_doc:
233            raise ValueError(f"Process document for {method_name} is missing required 'output_key' field")
234            
235        # Convert firestore document to process config
236        process_config = []
237        
238        # Check if we have a single step document (without s1, s2, etc.)
239        if 'type' in process_doc and 'action' in process_doc and 'params' in process_doc:
240            # Single step document - use it directly
241            config = {
242                'type': process_doc['type'],
243                'action': process_doc['action'],
244                'params': self._prepare_params(process_doc['params'], inputs),
245                'output_key': process_doc.get('output_key')
246            }
247            if 'additional_params' in process_doc:
248                config['additional_params'] = process_doc['additional_params']
249            process_config.append(config)
250        else:
251            # Multiple steps with s1, s2, etc.
252            steps = sorted(
253                [(k, v) for k, v in process_doc.items() if k.startswith('s')],
254                key=lambda x: int(x[0][1:])
255            )
256            
257            if not steps:
258                raise ValueError(f"No steps found in process document for {method_name}")
259            
260            # Build process config
261            for _, step in steps:
262                # Validate required step fields
263                required_fields = {'type', 'action', 'params'}
264                missing_fields = required_fields - set(step.keys())
265                if missing_fields:
266                    raise ValueError(f"Step is missing required fields: {missing_fields}")
267                    
268                config = {
269                    'type': step['type'],
270                    'action': step['action'],
271                    'params': self._prepare_params(step['params'], inputs),
272                    'output_key': step.get('output_key')
273                }
274                if 'additional_params' in step:
275                    config['additional_params'] = step['additional_params']
276                process_config.append(config)
277        
278        # Execute process
279        results = self.data_manipulator.execute_process(process_config)
280        
281        # Return the final output using the specified output key
282        output_key = process_doc['output_key']
283        if output_key not in results:
284            raise ValueError(f"Output key '{output_key}' not found in process results")
285            
286        return results[output_key]

288    def _prepare_params(self, params: List, inputs: Dict) -> List:
289        """
290        Prepare parameters by resolving references to input values.
291
292        :param params: List of parameter definitions.
293        :param inputs: Dictionary of input values.
294        :return: List of resolved parameter values.
295        :raises ValueError: If required inputs are missing or invalid.
296        """
297        def get_nested_value(data: Dict, path: str) -> Any:
298            """
299            Get a value from nested dictionary using dot notation.
300            Example: get_nested_value({'inputs': {'artifact': 'value'}}, 'inputs.artifact')
301            """
302            current = data
303            parts = path.split('.')
304            
305            for part in parts:
306                if isinstance(current, dict) and part in current:
307                    current = current[part]
308                else:
309                    raise ValueError(f"Required input '{path}' not found in structure: {data}")
310            
311            return current
312
313        prepared = []
314        for param in params:
315            if isinstance(param, str) and param.startswith('@'):
316                input_key = param[1:]  # Remove the '@' symbol
317                try:
318                    # First try direct access
319                    if input_key in inputs:
320                        prepared.append(inputs[input_key])
321                    # Then try nested access in 'inputs'
322                    elif 'inputs' in inputs:
323                        prepared.append(get_nested_value(inputs, f'inputs.{input_key}'))
324                    else:
325                        raise ValueError(f"Required input '{input_key}' not found")
326                except (KeyError, TypeError):
327                    raise ValueError(f"Required input '{input_key}' not found or has invalid format")
328            else:
329                prepared.append(param)
330                
331        return prepared

333    def execute_scenario(self, scenario_name: str, initial_input: Dict) -> Dict:
334        """
335        Execute a scenario by performing a sequence of steps.
336
337        :param scenario_name: Name of the scenario (Firestore collection).
338        :param initial_input: Initial input data for the scenario.
339        :return: Cache containing intermediate and final outputs.
340        :raises ValueError: If the scenario or steps are invalid.
341        """
342        # Read all documents from the collection and sort by document ID
343        scenario_docs = self.read(collection=scenario_name)
344        if not scenario_docs:
345            raise ValueError(f"Scenario '{scenario_name}' not found")
346        
347        # Sort documents by their step number (s1, s2, s3, etc.)
348        sorted_docs = sorted(
349            scenario_docs,
350            key=lambda x: int(x.get('step_number', x.get('__name__', '0').replace('s', '')))
351        )
352        
353        print(f"Found {len(sorted_docs)} steps")
354        
355        # Parse steps into Step objects
356        steps = []
357        for doc in sorted_docs:
358            step_data = {
359                'actor': doc.get('actor'),
360                'action': doc.get('action'),
361                'kwargs': doc.get('kwargs', {}),
362                'output_name': doc.get('output_name')
363            }
364            steps.append(Step(**step_data))
365        
366        print(f"Steps to execute: {steps}")
367                
368        # Initialize cache with input
369        self.cache = {"input": initial_input}
370        
371        # Execute each step
372        for i, step in enumerate(steps, 1):
373            try:
374                # Replace any cache references in kwargs
375                processed_kwargs = self._process_kwargs(step.kwargs)
376                
377                # Execute the appropriate action based on step type
378                result = None
379                if step.action == "read":
380                    result = self.read(**processed_kwargs)
381                elif step.action == "write":
382                    result = self.write(**processed_kwargs)
383                elif step.action == "update":

385                elif step.action == "delete":
386                    result = self.delete(**processed_kwargs)
387                elif step.action == "manipulate":

392                elif step.action == "send":

394                        service=processed_kwargs.pop('service'),
395                        endpoint=processed_kwargs.pop('endpoint'),
396                        data=processed_kwargs.get('data'),
397                        api_key=processed_kwargs.get('api_key')
398                    )
399                else:
400                    raise ValueError(f"Unknown action: {step.action}")
401                
402                # Store result in cache if output_name is specified
403                if step.output_name:
404                    self.cache[step.output_name] = result
405
406                print(f"Step {i} executed successfully: {result}")
407                
408                # Store step execution status
409                self.cache[f"step_{i}_status"] = "success"
410                
411            except Exception as e:
412                # Store error information in cache
413                self.cache[f"step_{i}_status"] = "error"
414                self.cache[f"step_{i}_error"] = str(e)

419    def _process_kwargs(self, kwargs: Dict) -> Dict:
420        """
421        Process kwargs dictionary to replace cache references with actual values.
422        Supports nested cache references using dot notation (e.g., 'input.user_message')
423        
424        Args:
425            kwargs (Dict): Original kwargs dictionary
426            
427        Returns:
428            Dict: Processed kwargs with cache values substituted
429        """
430        def get_nested_value(cache: Dict, key_path: str):
431            """Helper function to get nested values using dot notation"""
432            keys = key_path.split('.')
433            value = cache
434            for k in keys:
435                if isinstance(value, dict) and k in value:
436                    value = value[k]
437                else:
438                    raise ValueError(f"Cache reference '{key_path}' not found")
439            return value
440
441        processed = {}
442        for key, value in kwargs.items():
443            if isinstance(value, str) and value.startswith("$"):
444                # Remove $ and get the cache key
445                cache_key = value[1:]
446                try:
447                    processed[key] = get_nested_value(self.cache, cache_key)
448                except ValueError as e:
449                    raise ValueError(f"Cache reference '{cache_key}' not found")
450            elif isinstance(value, dict):
451                # Recursively process nested dictionaries
452                processed[key] = self._process_kwargs(value)
453            elif isinstance(value, list):
454                # Process lists that might contain cache references
455                processed[key] = [
456                    self._process_kwargs({0: v})[0] if isinstance(v, (str, dict))
457                    else v for v in value
458                ]
459            else:
460                processed[key] = value
461                
462        return processed

289    def _prepare_params(self, params: List, inputs: Dict) -> List:
290        """
291        Prepare parameters by resolving references to input values.
292
293        :param params: List of parameter definitions.
294        :param inputs: Dictionary of input values.
295        :return: List of resolved parameter values.
296        :raises ValueError: If required inputs are missing or invalid.
297        """
298        def get_nested_value(data: Dict, path: str) -> Any:
299            """
300            Get a value from nested dictionary using dot notation.
301            Example: get_nested_value({'inputs': {'artifact': 'value'}}, 'inputs.artifact')
302            """
303            current = data
304            parts = path.split('.')
305            
306            for part in parts:
307                if isinstance(current, dict) and part in current:
308                    current = current[part]
309                else:
310                    raise ValueError(f"Required input '{path}' not found in structure: {data}")
311            
312            return current
313
314        prepared = []
315        for param in params:
316            if isinstance(param, str) and param.startswith('@'):
317                input_key = param[1:]  # Remove the '@' symbol
318                try:
319                    # First try direct access
320                    if input_key in inputs:
321                        prepared.append(inputs[input_key])
322                    # Then try nested access in 'inputs'
323                    elif 'inputs' in inputs:
324                        prepared.append(get_nested_value(inputs, f'inputs.{input_key}'))
325                    else:
326                        raise ValueError(f"Required input '{input_key}' not found")
327                except (KeyError, TypeError):
328                    raise ValueError(f"Required input '{input_key}' not found or has invalid format")
329            else:
330                prepared.append(param)
331                
332        return prepared

420    def _process_kwargs(self, kwargs: Dict) -> Dict:
421        """
422        Process kwargs dictionary to replace cache references with actual values.
423        Supports nested cache references using dot notation (e.g., 'input.user_message')
424        
425        Args:
426            kwargs (Dict): Original kwargs dictionary
427            
428        Returns:
429            Dict: Processed kwargs with cache values substituted
430        """
431        def get_nested_value(cache: Dict, key_path: str):
432            """Helper function to get nested values using dot notation"""
433            keys = key_path.split('.')
434            value = cache
435            for k in keys:
436                if isinstance(value, dict) and k in value:
437                    value = value[k]
438                else:
439                    raise ValueError(f"Cache reference '{key_path}' not found")
440            return value
441
442        processed = {}
443        for key, value in kwargs.items():
444            if isinstance(value, str) and value.startswith("$"):
445                # Remove $ and get the cache key
446                cache_key = value[1:]
447                try:
448                    processed[key] = get_nested_value(self.cache, cache_key)
449                except ValueError as e:
450                    raise ValueError(f"Cache reference '{cache_key}' not found")
451            elif isinstance(value, dict):
452                # Recursively process nested dictionaries
453                processed[key] = self._process_kwargs(value)
454            elif isinstance(value, list):
455                # Process lists that might contain cache references
456                processed[key] = [
457                    self._process_kwargs({0: v})[0] if isinstance(v, (str, dict))
458                    else v for v in value
459                ]
460            else:
461                processed[key] = value
462                
463        return processed
